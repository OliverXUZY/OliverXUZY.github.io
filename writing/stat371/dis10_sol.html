#Bootrap Function 
bootstrap = function(x, n.boot) {
  n = length(x)
  x.bar <- mean(x)
  t.hat <- numeric(n.boot) # create vector of length n.boot zeros
  for(i in 1:n.boot) {
    x.star <- sample(x, size=n, replace=TRUE)
    x.bar.star <- mean(x.star)
    s.star <- sd(x.star)
    t.hat[i] <- (x.bar.star - x.bar) / (s.star / sqrt(n))
}
  return(t.hat)
}
